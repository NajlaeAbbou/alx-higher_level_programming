def add_integer(a, b=98):
    """
    Adds two integers.

    This function takes two arguments, 'a' and 'b', which should be integers or floats.
    If 'a' or 'b' is a float, they will be cast to integers before performing the addition.
    
    Parameters:
    a (int or float): The first number to be added.
    b (int or float, optional): The second number to be added. Default is 98.
    
    Returns:
    int: The sum of 'a' and 'b'.
    
    Raises:
    TypeError: If 'a' or 'b' is neither an integer nor a float.
    
    Examples:
    >>> add_integer(5, 3)
    8
    >>> add_integer(10.5, 7)
    17
    >>> add_integer(2, 4.5)
    6
    >>> add_integer(7.2, 3.8)
    11
    >>> add_integer("1", 2)
    Traceback (most recent call last):
        ...
    TypeError: a must be an integer or b must be an integer
    >>> add_integer(3, "4")
    Traceback (most recent call last):
        ...
    TypeError: a must be an integer or b must be an integer
    """
    if not (isinstance(a, int) or isinstance(a, float)) or not (isinstance(b, int) or isinstance(b, float)):
        raise TypeError("a must be an integer or b must be an integer")

    if isinstance(a, float):
        a = int(a)
    if isinstance(b, float):
        b = int(b)

    return a + b
